    a. Initial guess x0, x1
    b. Tolerable Error e
    c. Maximum Iteration N

5. Do
    If f(x0) = f(x1)
        Print "Mathematical Error"
        Stop
    End If

    x2 = x1 - (x1 - x0) * f(x1) / ( f(x1) - f(x0) )
    x0 = x1
    x1 = x2
    step = step + 1
    If step > N
        Print "Not Convergent"
        Stop
    End If

    While abs f(x2) > e
